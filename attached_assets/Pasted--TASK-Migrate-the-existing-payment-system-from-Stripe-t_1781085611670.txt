### 1.3 Backend — Replace Payment Routes

Delete the existing Stripe payment route file and replace it with the following two endpoints:

**POST `/api/payments/create-order`**

- Accepts: `{ booking_id, total_amount }`
- Creates a Razorpay order:
  
  ```js
  const order = await razorpay.orders.create({
    amount: Math.round(total_amount * 100), // paise
    currency: "inr",
    receipt: String(booking_id),
  });
  ```
- Returns: `{ order_id, amount, currency, key_id: process.env.RAZORPAY_KEY_ID }`

**POST `/api/payments/verify-order`**

- Accepts: `{ razorpay_order_id, razorpay_payment_id, razorpay_signature, booking_id }`
- Verify signature using HMAC SHA256:
  
  ```js
  const crypto = require("crypto");
  const body = razorpay_order_id + "|" + razorpay_payment_id;
  const expected = crypto
    .createHmac("sha256", process.env.RAZORPAY_KEY_SECRET)
    .update(body)
    .digest("hex");
  const isValid = expected === razorpay_signature;
  ```
- If valid:
  - Update booking `status` to `'confirmed'` in the database
  - Trigger WhatsApp confirmation message (see Part 2)
  - Trigger Resend email confirmation (keep existing email logic)
  - Return: `{ success: true, booking_id }`
- If invalid: return `{ success: false }` with status 400

**Remove:**

- `POST /api/payments/create-intent` (Stripe endpoint)
- Any existing Stripe webhook route

### 2.2 Create WhatsApp Service

Create file: `artifacts/api-server/src/services/whatsappService.js`

```js
const { Client, LocalAuth } = require('whatsapp-web.js');
const qrcode = require('qrcode-terminal');

let client;
let isReady = false;

function initWhatsApp() {
  try {
    client = new Client({
      authStrategy: new LocalAuth(),
      puppeteer: {
        args: ['--no-sandbox', '--disable-setuid-sandbox'],
      }
    });

    client.on('qr', (qr) => {
      console.log('\n========================================');
      console.log('SCAN THIS QR CODE IN WHATSAPP:');
      console.log('========================================\n');
      qrcode.generate(qr, { small: true });
    });

    client.on('ready', () => {
      isReady = true;
      console.log('WhatsApp client is ready.');
    });

    client.on('disconnected', () => {
      isReady = false;
      console.log('WhatsApp client disconnected.');
    });

    client.initialize();
  } catch (err) {
    console.warn('WhatsApp init failed — skipping:', err.message);
  }
}

async function sendBookingConfirmation({ phone, guestName, bookingId, rooms, checkIn, checkOut, totalPrice }) {
  if (!isReady || !client) {
    console.warn('WhatsApp not ready — skipping message.');
    return;
  }
  try {
    // Strip leading 0, prepend India country code
    const cleaned = phone.replace(/\D/g, '').replace(/^0/, '');
    const number = `91${cleaned}@c.us`;

    const message =
      `🏨 *Booking Confirmed!*\n\n` +
      `Hi ${guestName},\n` +
      `Your booking has been confirmed.\n\n` +
      `📋 *Booking ID:* ${bookingId}\n` +
      `🛏️ *Rooms:* ${rooms.join(', ')}\n` +
      `📅 *Check-in:* ${checkIn}\n` +
      `📅 *Check-out:* ${checkOut}\n` +
      `💰 *Total Paid:* ₹${Number(totalPrice).toLocaleString('en-IN')}\n\n` +
      `Thank you for your booking!`;

    await client.sendMessage(number, message);
    console.log(`WhatsApp confirmation sent to ${number}`);
  } catch (err) {
    console.warn('WhatsApp send failed — skipping:', err.message);
  }
}

function getWhatsAppStatus() {
  return isReady;
}

module.exports = { initWhatsApp, sendBookingConfirmation, getWhatsAppStatus };
```

## IMPORTANT RULES

1. **Never crash the server** over WhatsApp. Every `whatsapp-web.js` call must be in a try-catch.
1. **Keep Resend email** — WhatsApp is supplementary, not a replacement. Send both.
1. **LocalAuth persists the session** to disk so the QR code only needs to be scanned once. Do not use `NoAuth`.
1. **Phone number format:** strip all non-digits, remove leading 0, prepend `91`, append `@c.us`.
1. **Currency:** all prices display as INR using `en-IN` locale. Amount sent to Razorpay is always in paise (×100).
1. **Do not touch** the database schema, seed data, admin login, or any page unrelated to payments.
